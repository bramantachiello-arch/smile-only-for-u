<script>
/* ====== PERSONALIZE: ganti nama dan baris pesan di sini ====== */
const loverName = "Aisya"; // ubah sesuai nama pacarmu
const messages = [
  `Hai ${loverName} ‚Äî kamu klik satu kali. Jadi aku ingat: cuma butuh satu senyum buat bikin hari jadi lebih ringan.`,
  `Klik lagi ya? Aku jadi inget hal kecil: suaramu pas ketawa itu favoritku üòå.`,
  `Oke, tiga kali. Aku serius nih: aku suka cara kamu nengok dunia‚Äîtenang tapi kuat.`,
  `Empat kali? Wah, kamu serius bikin aku mikir, "gue beruntung punya kamu."`,
  `Kelima klik ‚Äî rahasiaku: aku pengin jadi alasan kamu tersenyum tiap hari.`
];
/* ========================================================== */

let count = 0;
const heartBtn = document.getElementById('heartBtn');
const counterText = document.getElementById('counterText');
const msg = document.getElementById('msg');
const confettiBox = document.getElementById('confetti');
const card = document.getElementById('card');
const resetBtn = document.getElementById('resetBtn');
const hintBtn = document.getElementById('hintBtn');

function setCounterText(){
  counterText.textContent = `Klik hati ini ‚Ä¢ ${count} kali`;
}

function showTypingThen(text, cb){
  msg.innerHTML = `<span class="typing">...</span>`;
  // simulate typing time by length
  const t = Math.min(1400, 200 + text.length*25);
  setTimeout(()=>{ msg.innerHTML = text; if(cb) cb(); }, t);
}

function tinyBounce(){
  card.classList.add('pulse');
  setTimeout(()=> card.classList.remove('pulse'), 900);
}

function spawnConfetti(amt){
  for(let i=0;i<amt;i++){
    const el = document.createElement('div');
    el.className = 'c';
    el.textContent = Math.random()>0.6 ? 'üíõ' : 'üíó';
    const size = 12 + Math.random()*18;
    el.style.left = (10 + Math.random()*80) + '%';
    el.style.top = (-10 - Math.random()*20) + '%';
    el.style.fontSize = size + 'px';
    confettiBox.appendChild(el);
    // animate
    const fall = 2000 + Math.random()*1800;
    el.animate([
      { transform: `translateY(0) rotate(0)`, opacity:0 },
      { transform: `translateY(${window.innerHeight + 100}px) rotate(${Math.random()*360}deg)`, opacity:1 }
    ], { duration: fall, easing:'cubic-bezier(.2,.7,.2,1)'});
    // remove after
    setTimeout(()=> el.remove(), fall + 200);
  }
}

/* main interaction */
heartBtn.addEventListener('click', ()=>{
  count++;
  setCounterText();
  tinyBounce();
  heartBtn.style.transform = 'scale(1.06)';
  setTimeout(()=> heartBtn.style.transform = '', 140);

  // staged messages
  if(count <= messages.length){
    showTypingThen(messages[count-1], ()=>{
      // small confetti on second and last click
      if(count === 2) spawnConfetti(8);
      if(count === messages.length) {
        spawnConfetti(18);
        // reveal special playful line after a moment
        setTimeout(()=> {
          msg.innerHTML = `Oh ya... sebelum lupa ‚Äî coba deh bilang "yang" (sekali aja). Aku janji aku jawab <b>sayang</b>. ü§≠`;
        }, 800);
      }
    });
  } else {
    // beyond messages: playful flirty replies rotate
    const extras = [
      `Aku jadi kepikiran buat ngajak kamu makan bareng nanti. Siap gak? üòè`,
      `Kamu bener-bener lucu kalo senyum. Aku gak bohong kok.`,
      `Nanti kalau ketemu, aku traktir es krim ‚Äî karena kamu pantas yang manis.`,
      `Klik lagi? Atau kamu udahan dulu, sayang?`
    ];
    const pick = extras[(count - messages.length -1) % extras.length];
    showTypingThen(pick);
    if(count % 4 === 0) spawnConfetti(10);
  }
});

/* reset */
resetBtn.addEventListener('click', ()=> {
  count = 0;
  setCounterText();
  msg.textContent = `Halo, aku di sini. Klik sekali aja dulu, biar aku tunjukin sesuatu. :)`;
});

/* hint */
hintBtn.addEventListener('click', ()=>{
  showTypingThen(`Kalau kamu peka: coba bilang "yang" di chat nanti. Jangan khawatir, aku gak bakal GR. üòå`);
});

/* small easter: if they type 'yang' in prompt (optional) */
document.addEventListener('keydown', (e)=>{
  // secret: if user types "y" then "a" then "n" then "g" quickly
  // not necessary; keep simple: double-click heart quickly triggers playful reveal
});

/* small UX: double-click reveals short note */
heartBtn.addEventListener('dblclick', ()=> {
  showTypingThen(`Dbl klik? Oke: aku beneran sayang sama kamu. Bukan main-main. üíå`);
  spawnConfetti(14);
});

/* init */
setCounterText();
